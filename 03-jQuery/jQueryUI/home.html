  <script>
    $(document).ready(function () {
      let position = $("#position");
      let offset = $("#offset");
      let draggable = $("#draggable");
      draggable.draggable({
        drag: function (event, ui) {
          position.html("CSS position: " + JSON.stringify(ui.position));
          offset.html("Offset position: " + JSON.stringify(ui.offset));
        }
      })
    });
  </script>
